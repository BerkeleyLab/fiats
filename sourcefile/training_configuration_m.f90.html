module training_configuration_m
  use sourcery_m, only : string_t, file_t

      implicit none
      class(training_configuration_t), intent(in) :: self
      real(rkind) rate
    end function
 
    pure module function differentiable_activation_strategy(self) result(strategy)
      implicit none
      class(training_configuration_t), intent(in) :: self
