module swish_m
  use differentiable_activation_strategy_m, only : differentiable_activation_strategy_t
  use kind_parameters_m, only : rkind
  use string_m, only : string_t
  implicit none

  private
  public :: swish_t 

  type, extends(differentiable_activation_strategy_t) :: swish_t
  contains
     procedure, nopass :: activation
     procedure, nopass :: activation_derivative
     procedure, nopass :: function_name
  end type

  interface

    elemental module function activation(x) result(y)
      implicit none
      real(rkind), intent(in) :: x
      real(rkind) y
    end function

    elemental module function activation_derivative(x) result(y)
      implicit none
      real(rkind), intent(in) :: x
      real(rkind) y
    end function

    elemental module function function_name() result(string)
      implicit none
      type(string_t) string
    end function

  end interface

end module swish_m
